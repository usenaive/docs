<ParamField header="X-Company-Id" type="string" required>
  The company the deliverable belongs to.
</ParamField>

<ParamField path="id" type="string" required>
  The deliverable ID to retrieve.
</ParamField>

<CodeGroup>
```bash curl
curl https://api.usenaive.ai/v1/deliverables/del_abc123 \
  -H "Authorization: Bearer nv_pk_your_key_here" \
  -H "X-Company-Id: company_abc123"
```

```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/deliverables/del_abc123",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
)
data = response.json()
```


<ResponseField name="id" type="string">
  Unique identifier for the deliverable.
</ResponseField>

<ResponseField name="title" type="string">
  Display title of the deliverable.
</ResponseField>

<ResponseField name="type" type="string">
  Deliverable type (e.g. `email_draft`, `blog_post`).
</ResponseField>

<ResponseField name="folder" type="string">
  Folder category the deliverable belongs to.
</ResponseField>

<ResponseField name="status" type="string">
  Review status: `pending_review`, `approved`, or `rejected`.
</ResponseField>

<ResponseField name="content" type="string">
  Full content of the deliverable.
</ResponseField>

<ResponseField name="preview" type="string">
  Rendered preview URL or HTML snippet, or `null`.
</ResponseField>

<ResponseField name="file_path" type="string">
  Path to an attached file or asset, or `null`.
</ResponseField>

<ResponseField name="employee_id" type="string">
  The employee that produced this deliverable.
</ResponseField>

<ResponseField name="task_id" type="string">
  Linked task ID, or `null`.
</ResponseField>

<ResponseField name="campaign_id" type="string">
  Linked campaign ID, or `null`.
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the deliverable was created.
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of the last update.
</ResponseField>
