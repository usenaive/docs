---
title: "Submit Deliverable"
description: "Submit a new deliverable for review."
api: "POST /v1/deliverables"
---

<Info>This endpoint is **coming soon** and is not yet available for use.</Info>

<ParamField header="X-Company-Id" type="string" required>
  The company to submit the deliverable under.
</ParamField>

<ParamField body="title" type="string" required>
  Display title for the deliverable.
</ParamField>

<ParamField body="type" type="string" required>
  Deliverable type. One of `email_draft`, `social_post`, `blog_post`, `landing_page`, `strategy_doc`, `report`, `image_set`, `video`, `presentation`, `code`, `design`, `document`, or `other`.
</ParamField>

<ParamField body="folder" type="string" required>
  Folder to organize the deliverable into (e.g. `emails`, `blog`, `social`, `assets`).
</ParamField>

<ParamField body="content" type="string" required>
  The full content of the deliverable.
</ParamField>

<ParamField body="employee_id" type="string" required>
  The employee (agent) that produced this deliverable.
</ParamField>

<ParamField body="preview" type="string">
  Optional rendered preview URL or HTML snippet.
</ParamField>

<ParamField body="file_path" type="string">
  Optional path to an attached file or asset.
</ParamField>

<ParamField body="task_id" type="string">
  Link this deliverable to a specific task.
</ParamField>

<ParamField body="campaign_id" type="string">
  Link this deliverable to a campaign.
</ParamField>

<CodeGroup>
```bash curl
curl -X POST https://api.usenaive.ai/v1/deliverables \
  -H "Authorization: Bearer nv_pk_your_key_here" \
  -H "X-Company-Id: company_abc123" \
  -H "Content-Type: application/json" \
  -d '{
    "title": "Weekly Newsletter Draft",
    "type": "email_draft",
    "folder": "emails",
    "content": "Hi there! Here is what happened this week at Acme Corp...",
    "employee_id": "emp_abc123",
    "task_id": "task_xyz789"
  }'
```

```python Python
import requests

response = requests.post(
    "https://api.usenaive.ai/v1/deliverables",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
    json={
        "title": "Weekly Newsletter Draft",
        "type": "email_draft",
        "folder": "emails",
        "content": "Hi there! Here is what happened this week at Acme Corp...",
        "employee_id": "emp_abc123",
        "task_id": "task_xyz789",
    },
)
data = response.json()
```

<ResponseField name="id" type="string">
  Unique identifier for the newly created deliverable.
</ResponseField>

<ResponseField name="title" type="string">
  Display title of the deliverable.
</ResponseField>

<ResponseField name="type" type="string">
  Deliverable type.
</ResponseField>

<ResponseField name="folder" type="string">
  Folder the deliverable was placed in.
</ResponseField>

<ResponseField name="status" type="string">
  Initial review status (always `pending_review`).
</ResponseField>

<ResponseField name="content" type="string">
  Full content of the deliverable.
</ResponseField>

<ResponseField name="employee_id" type="string">
  The employee that produced this deliverable.
</ResponseField>

<ResponseField name="task_id" type="string">
  Linked task ID, or `null`.
</ResponseField>

<ResponseField name="campaign_id" type="string">
  Linked campaign ID, or `null`.
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the deliverable was created.
</ResponseField>
