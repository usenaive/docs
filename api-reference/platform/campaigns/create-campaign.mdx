<ParamField header="X-Company-Id" type="string" required>
  The company to create the campaign in.
</ParamField>

<ParamField body="title" type="string" required>
  Title for the campaign.
</ParamField>

<ParamField body="description" type="string">
  Description of the campaign.
</ParamField>

<ParamField body="status" type="string">
  Initial campaign status.
</ParamField>

<ParamField body="target_date" type="string">
  ISO 8601 target completion date.
</ParamField>

<ParamField body="goal_id" type="string">
  ID of the goal this campaign supports.
</ParamField>

<ParamField body="plan" type="string">
  Detailed plan text for the campaign.
</ParamField>

<ParamField body="success_criteria" type="string">
  Criteria for measuring campaign success.
</ParamField>

<ParamField body="lead_employee_id" type="string">
  Employee ID of the campaign lead.
</ParamField>

<ParamField body="planned_task_count" type="number">
  Number of planned tasks in the campaign.
</ParamField>

<ParamField body="phases" type="array">
  Optional array of campaign phases. Each phase has a `title` and `description`.
</ParamField>

import requests

response = requests.post(
    "https://api.usenaive.ai/v1/campaigns",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
    json={
        "title": "Q2 Product Launch",
        "description": "Launch the new product line with a multi-channel campaign.",
        "target_date": "2026-06-30T00:00:00Z",
        "goal_id": "goal_abc123",
        "lead_employee_id": "emp_abc123",
        "planned_task_count": 12,
        "phases": [
            {"title": "Pre-launch", "description": "Build hype and teasers"},
            {"title": "Launch Week", "description": "Coordinated launch across channels"},
        ],
    },
)
data = response.json()
