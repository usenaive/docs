```bash curl
curl https://api.usenaive.ai/v1/campaigns \
  -H "Authorization: Bearer nv_pk_your_key_here" \
  -H "X-Company-Id: company_abc123"
```

```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/campaigns",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
)
data = response.json()
```

<ResponseField name="campaigns" type="array">
  Array of campaign objects.
</ResponseField>

<ResponseField name="campaigns[].id" type="string">
  Unique campaign identifier.
</ResponseField>

<ResponseField name="campaigns[].title" type="string">
  Campaign title.
</ResponseField>

<ResponseField name="campaigns[].status" type="string">
  Current campaign status.
</ResponseField>

<ResponseField name="campaigns[].description" type="string">
  Campaign description.
</ResponseField>

<ResponseField name="campaigns[].target_date" type="string">
  ISO 8601 target completion date.
</ResponseField>

<ResponseField name="campaigns[].goal_id" type="string">
  ID of the associated goal, or `null`.
</ResponseField>

<ResponseField name="campaigns[].lead_employee_id" type="string">
  Employee ID of the campaign lead.
</ResponseField>

<ResponseField name="campaigns[].planned_task_count" type="number">
  Number of planned tasks in the campaign.
</ResponseField>

<ResponseField name="campaigns[].created_at" type="string">
  ISO 8601 timestamp of when the campaign was created.
</ResponseField>
