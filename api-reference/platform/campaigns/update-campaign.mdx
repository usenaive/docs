```bash curl
curl -X PATCH https://api.usenaive.ai/v1/campaigns/camp_abc123 \
  -H "Authorization: Bearer nv_pk_your_key_here" \

  -H "Content-Type: application/json" \
  -d '{
```

```python Python
import requests

response = requests.patch(

    },
)
data = response.json()
```
