```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/employees",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
)
data = response.json()
```

<ResponseField name="employees" type="array">
  Array of employee objects.
</ResponseField>

<ResponseField name="employees[].paperclipId" type="string">
  Unique Naive identifier for the employee.
</ResponseField>

<ResponseField name="employees[].agentId" type="string">
  Internal agent identifier.
</ResponseField>

<ResponseField name="employees[].name" type="string">
  Employee display name.
</ResponseField>

<ResponseField name="employees[].title" type="string">
  Job title (e.g., "Content Writer").
</ResponseField>

<ResponseField name="employees[].role" type="string">
  Role slug describing the employee's function.
</ResponseField>

<ResponseField name="employees[].capabilities" type="array">
  List of primitive categories the employee can use.
</ResponseField>

<ResponseField name="employees[].reportsTo" type="string">
  `paperclipId` of the employee's manager, or `null` if top-level.
</ResponseField>

<ResponseField name="employees[].status" type="string">
  Current status: `active`, `paused`, or `decommissioned`.
</ResponseField>
