<ParamField header="X-Company-Id" type="string" required>
  The company to create the goal in.
</ParamField>

<ParamField body="title" type="string" required>
  Title for the goal.
</ParamField>

<ParamField body="description" type="string">
  Detailed description of the goal.
</ParamField>

<ParamField body="level" type="string">
  Goal level in the hierarchy (e.g., "company", "team", "individual").
</ParamField>

<ParamField body="status" type="string">
  Initial status. Defaults to `active`.
</ParamField>

<ParamField body="parent_id" type="string">
  ID of a parent goal to nest this goal under.
</ParamField>

<ParamField body="owner_employee_id" type="string">
  Employee ID of the goal owner.
</ParamField>

```bash curl
curl -X POST https://api.usenaive.ai/v1/goals \
  -H "Authorization: Bearer nv_pk_your_key_here" \

  -H "Content-Type: application/json" \
  -d '{
```

```python Python
import requests

response = requests.post(
    "https://api.usenaive.ai/v1/goals",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
    json={
        "title": "Increase brand awareness by 30%",
        "description": "Drive measurable brand awareness growth across all digital channels.",
        "level": "company",
    },
)
data = response.json()
```
