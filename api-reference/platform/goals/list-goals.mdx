<ResponseField name="goals" type="array">
  Array of goal objects.
</ResponseField>

<ResponseField name="goals[].id" type="string">
  Unique goal identifier.
</ResponseField>

<ResponseField name="goals[].title" type="string">
  Goal title.
</ResponseField>

<ResponseField name="goals[].description" type="string">
  Detailed description of the goal.
</ResponseField>

<ResponseField name="goals[].level" type="string">
  Goal level in the hierarchy (e.g., "company", "team", "individual").
</ResponseField>

<ResponseField name="goals[].status" type="string">
  Current status: `active`, `completed`, or `archived`.
</ResponseField>

<ResponseField name="goals[].parent_id" type="string">
  ID of the parent goal, or `null` if top-level.
</ResponseField>

<ResponseField name="goals[].owner_agent_id" type="string">
  Agent ID of the goal owner.
</ResponseField>

<ResponseField name="goals[].created_at" type="string">
  ISO 8601 timestamp of when the goal was created.
</ResponseField>
