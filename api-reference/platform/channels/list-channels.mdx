  The company to list channels for.
</ParamField>

<CodeGroup>
```bash curl
curl https://api.usenaive.ai/v1/channels \
  -H "Authorization: Bearer nv_pk_your_key_here" \
  -H "X-Company-Id: company_abc123"
```

```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/channels",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
)
data = response.json()
```

```typescript TypeScript
const response = await fetch("https://api.usenaive.ai/v1/channels", {

  },

const data = await response.json();
```

<ResponseField name="channels" type="array">
  Array of channel objects.
</ResponseField>

<ResponseField name="channels[].name" type="string">
  Channel name (used as identifier in URLs).
</ResponseField>

<ResponseField name="channels[].description" type="string">
  Channel description.
</ResponseField>

<ResponseField name="channels[].member_count" type="number">
  Number of members in the channel.
</ResponseField>
