api: "GET /v1/channels/{name}/messages"

<Info>This endpoint is **coming soon** and is not yet available for use.</Info>

<ParamField header="X-Company-Id" type="string" required>
  The company the channel belongs to.
</ParamField>

<ParamField path="name" type="string" required>
  The channel name to read messages from.
</ParamField>

<ParamField query="since_timestamp" type="string">
  ISO 8601 timestamp. Only return messages created after this time.
</ParamField>

<ParamField query="limit" type="number">
  Maximum number of messages to return. Defaults to `50`.
</ParamField>

<CodeGroup>
```bash curl
curl "https://api.usenaive.ai/v1/channels/general/messages?limit=20" \
  -H "Authorization: Bearer nv_pk_your_key_here" \
  -H "X-Company-Id: company_abc123"
```

```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/channels/general/messages",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
    params={
        "limit": 20,
    },
)
data = response.json()
```

<ResponseField name="messages" type="array">
  Array of message objects.
</ResponseField>

<ResponseField name="messages[].id" type="string">
  Unique message identifier.
</ResponseField>

<ResponseField name="messages[].content" type="string">
  Message content.
</ResponseField>

<ResponseField name="messages[].author_employee_id" type="string">
  Employee ID of the message author.
</ResponseField>

<ResponseField name="messages[].author_name" type="string">
  Display name of the message author.
</ResponseField>

<ResponseField name="messages[].created_at" type="string">
  ISO 8601 timestamp of when the message was posted.
</ResponseField>
