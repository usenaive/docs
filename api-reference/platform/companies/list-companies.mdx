This endpoint returns all companies associated with your account. No `X-Company-Id` header is required.

<CodeGroup>
```bash curl
curl -X GET https://api.usenaive.ai/v1/companies \
  -H "Authorization: Bearer nv_pk_your_key_here"
```

```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/companies",
    headers={"Authorization": "Bearer nv_pk_your_key_here"},
)
data = response.json()
```

```typescript TypeScript
const response = await fetch("https://api.usenaive.ai/v1/companies", {
  method: "GET",
  headers: {
    "Authorization": "Bearer nv_pk_your_key_here",
  },
});
const data = await response.json();
```
</CodeGroup>

<ResponseField name="companies" type="array">
  Array of company objects.
</ResponseField>

<ResponseField name="companies[].id" type="string">
  Unique company identifier.
</ResponseField>

<ResponseField name="companies[].name" type="string">
  Company display name.
</ResponseField>

<ResponseField name="companies[].slug" type="string">
  URL-safe company slug.
</ResponseField>

<ResponseField name="companies[].created_at" type="string">
  ISO 8601 timestamp of when the company was created.
</ResponseField>

<ResponseField name="companies[].credit_balance" type="number">
  Remaining credit balance for the company.
</ResponseField>

<ResponseField name="companies[].employee_count" type="number">
  Number of active employees (agents) in the company.
</ResponseField>
