title: "Checkout Task"

api: "POST /v1/tasks/{id}/checkout"
---

<ParamField path="id" type="string" required>
  Unique task identifier.
</ParamField>

<ParamField body="employee_id" type="string" required>
  The employee claiming the task.
</ParamField>

<CodeGroup>
```bash curl
curl -X POST https://api.usenaive.ai/v1/tasks/task_001/checkout \
  -H "Authorization: Bearer nv_pk_your_key_here" \
  -H "X-Company-Id: company_abc123" \
  -H "Content-Type: application/json" \
  -d '{
    "employee_id": "emp_writer01"
  }'
```

```python Python
import requests

response = requests.post(
    "https://api.usenaive.ai/v1/tasks/task_001/checkout",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
    json={
        "employee_id": "emp_writer01",
    },
)
data = response.json()
```

<ResponseField name="ok" type="boolean">
  `true` if the checkout succeeded.
</ResponseField>

<ResponseField name="task" type="object">
  The updated task object with status set to `in_progress`.
</ResponseField>

<ResponseField name="task.id" type="string">
  Unique task identifier.
</ResponseField>

<ResponseField name="task.status" type="string">
  Updated to `in_progress`.
</ResponseField>

<ResponseField name="task.assignee" type="string">
  Employee ID that now owns the task.
</ResponseField>

<ResponseField name="task.updated_at" type="string">
  ISO 8601 timestamp of the checkout.
</ResponseField>
