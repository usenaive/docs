curl -X GET https://api.usenaive.ai/v1/tasks/task_001 \

```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/tasks/task_001",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
)
data = response.json()
```

<ResponseField name="id" type="string">
  Unique task identifier.
</ResponseField>

<ResponseField name="title" type="string">
  Task title.
</ResponseField>

<ResponseField name="description" type="string">
  Task description.
</ResponseField>

<ResponseField name="status" type="string">
  Current status: `queued`, `in_progress`, `in_review`, `done`, or `blocked`.
</ResponseField>

<ResponseField name="priority" type="string">
  Priority level: `low`, `medium`, `high`, or `urgent`.
</ResponseField>

<ResponseField name="assignee" type="string">
  Assigned employee ID.
</ResponseField>

<ResponseField name="creator" type="string">
  ID of the user or agent that created the task.
</ResponseField>

<ResponseField name="goal_id" type="string">
  Associated goal identifier, if any.
</ResponseField>

<ResponseField name="campaign_id" type="string">
  Associated campaign identifier, if any.
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the task was created.
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of the last update.
</ResponseField>

<ResponseField name="comments" type="array">
  Array of comment objects attached to the task.
</ResponseField>

<ResponseField name="comments[].id" type="string">
  Unique comment identifier.
</ResponseField>

<ResponseField name="comments[].content" type="string">
  Comment body text.
</ResponseField>

<ResponseField name="comments[].employee_id" type="string">
  ID of the employee or user that posted the comment.
</ResponseField>

<ResponseField name="comments[].created_at" type="string">
  ISO 8601 timestamp of when the comment was posted.
</ResponseField>
