<ParamField query="assignee_employee_id" type="string">
  Filter tasks assigned to a specific employee.
</ParamField>

<ParamField query="status" type="string">
  Filter by task status. One of `queued`, `in_progress`, `in_review`, `done`, `blocked`.
</ParamField>

<CodeGroup>
```bash curl
curl -X GET "https://api.usenaive.ai/v1/tasks?status=in_progress" \
  -H "Authorization: Bearer nv_pk_your_key_here" \
  -H "X-Company-Id: company_abc123"
```

```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/tasks",
    headers={
        "Authorization": "Bearer nv_pk_your_key_here",
        "X-Company-Id": "company_abc123",
    },
    params={"status": "in_progress"},
)
data = response.json()
```

<ResponseField name="tasks" type="array">
  Array of task objects.
</ResponseField>

<ResponseField name="tasks[].id" type="string">
  Unique task identifier.
</ResponseField>

<ResponseField name="tasks[].title" type="string">
  Short title describing the task.
</ResponseField>

<ResponseField name="tasks[].description" type="string">
  Detailed task description.
</ResponseField>

<ResponseField name="tasks[].status" type="string">
  Current status: `queued`, `in_progress`, `in_review`, `done`, or `blocked`.
</ResponseField>

<ResponseField name="tasks[].priority" type="string">
  Priority level: `low`, `medium`, `high`, or `urgent`.
</ResponseField>

<ResponseField name="tasks[].assignee" type="string">
  Employee ID of the assigned agent.
</ResponseField>

<ResponseField name="tasks[].creator" type="string">
  Employee ID or user ID that created the task.
</ResponseField>

<ResponseField name="tasks[].goal_id" type="string">
  Associated goal identifier, if any.
</ResponseField>

<ResponseField name="tasks[].campaign_id" type="string">
  Associated campaign identifier, if any.
</ResponseField>

<ResponseField name="tasks[].created_at" type="string">
  ISO 8601 timestamp of when the task was created.
</ResponseField>

<ResponseField name="tasks[].updated_at" type="string">
  ISO 8601 timestamp of the last update.
</ResponseField>
