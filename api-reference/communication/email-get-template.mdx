---
title: "Get Template"
description: "Retrieve a single template by ID."
api: "GET /v1/email/templates/:templateId"
---

<ParamField path="templateId" type="string" required>
  The template ID.
</ParamField>

<CodeGroup>
```bash curl
curl https://api.usenaive.ai/v1/email/templates/tpl_abc123 \
  -H "Authorization: Bearer nv_sk_your_key_here"
```

```python Python
import requests

response = requests.get(
    "https://api.usenaive.ai/v1/email/templates/tpl_abc123",
    headers={"Authorization": "Bearer nv_sk_your_key_here"},
)
data = response.json()
```

```typescript TypeScript
const response = await fetch("https://api.usenaive.ai/v1/email/templates/tpl_abc123", {
  headers: { "Authorization": "Bearer nv_sk_your_key_here" },
});
const data = await response.json();
```


<ResponseField name="id" type="string">
  Template ID.
</ResponseField>

<ResponseField name="name" type="string">
  Template name.
</ResponseField>

<ResponseField name="subject" type="string">
  Default subject line.
</ResponseField>

<ResponseField name="status" type="string">
  Status: `draft` or `published`.
</ResponseField>
